# کنترل خروجی print

class Playlist:
    def init(self, name):
        self.name = name
        self.songs = []

    def add_song(self, song):
        self.songs.append(song)

    def show_songs(self):
        for song in self.songs:
            print(song)

my_playlist = Playlist("Favorites")
my_playlist.add_song("Bohemian Rhapsody")
my_playlist.show_songs()

class Person:
    def init(self, fname, lname):
        self.firstname = fname
        self.lastname = lname

    def printname(self):
        print(self.firstname, self.lastname)

x.printname()

    pass

x = Student("Mike", "Olsen")
x.printname()

# ساخت شیء از کلاس داخلی

class Outer:
    def init(self):
        self.name = "Emil"

    class Inner:
        def init(self, outer):
            self.outer = outer

        def display(self):
            print(self.outer.name)

            print("Storing")
